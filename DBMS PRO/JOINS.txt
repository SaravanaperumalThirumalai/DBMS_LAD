JOINS
CREATE TABLE Sailors (
sid NUMBER PRIMARY KEY,
sname VARCHAR2(14),
rating NUMBER,
age NUMBER );

CREATE TABLE Boats (
bid NUMBER PRIMARY KEY,
bname VARCHAR2(15),
color VARCHAR2(10) );

CREATE TABLE Reserves (
sid NUMBER,
bid NUMBER,
day DATE,
PRIMARY KEY (sid, bid, day),
FOREIGN KEY (sid) REFERENCES Sailors(sid),
FOREIGN KEY (bid) REFERENCES Boats(bid) );

SELECT S.sid, S.sname, B.bname, R.day
FROM Sailors S
INNER JOIN Reserves R ON S.sid = R.sid
INNER JOIN Boats B ON B.bid = R.bid;

SELECT S.sid, S.sname, B.bname, R.day
FROM Sailors S
LEFT JOIN Reserves R ON S.sid = R.sid
LEFT JOIN Boats B ON B.bid = R.bid;

SELECT S.sname, B.bname, R.day
FROM Sailors S
RIGHT JOIN Reserves R ON S.sid = R.sid
RIGHT JOIN Boats B ON B.bid = R.bid;

SELECT sid, S.sname, R.day
FROM Sailors S
NATURAL JOIN Reserves R;
